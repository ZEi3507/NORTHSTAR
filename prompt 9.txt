   File: src/lib/parsePostContent.ts

   export interface ParsedEntry {
     publicContent: string    // full text with [REDACTED] placeholder where tags were
     redactedContent: string  // extracted segments joined by \n---\n
     wordCount: number        // computed on FULL raw content including redacted text
   }

   export function parsePostContent(rawContent: string): ParsedEntry {
     - Find all [redact]...[/redact] blocks
     - Extract inner content into segments array
     - Replace each block in the original string with the literal text "[REDACTED]"
     - Join segments with "\n---\n" for redactedContent
     - Word count: strip the [redact] and [/redact] tags from rawContent,
       then split on whitespace and count non-empty tokens
     - Return { publicContent, redactedContent, wordCount }
   }

   Edge cases to handle:
   - Multiple [redact] blocks in one entry ✓
   - Nested content with line breaks inside [redact] blocks ✓
   - No [redact] tags at all → redactedContent is empty string, publicContent is full text ✓
   - Only [redact] content, no public text → publicContent is "[REDACTED]" ✓

   const MIN_WORD_COUNT = 150

const before = change.before.data()
const after = change.after.data()

APPROVAL PATH — fires when: before.status === 'pending' AND after.status === 'approved'

Trigger: Firestore onDelete — /archive/{postId}

    approvedPostIds: updatedApprovedIds

- Write updates to /scholars/{authorId}
